# secular_religion_head_trigger
# can_abolish_princely_elective_trigger
# landless_temporary_trigger

# Checks if a religion head title can have normal secular succession laws
secular_religion_head_trigger = {
	trigger_if = {
		limit = { is_primary_type_title = yes }

		controls_religion = yes
		is_landless_type_title = no
	}
}

can_abolish_elective_succession_trigger = {
	# NOTE: Do not combine these three. They are separate for the tooltip
	trigger_if = {
		limit = {
			has_dlc = "Conclave"

			OR = {
				has_law = succ_feudal_elective
				has_law = succ_hre_elective
			}
		}

		has_law = succession_voting_power_0
	}
	trigger_else_if = {
		limit = { has_law = succ_hre_elective }
		has_law = crown_authority_4
	}
	trigger_if = {
		limit = { has_law = succ_hre_elective }
		has_law = centralization_4
	}
}

# Used to block succession laws for landless revolts, adventurers, etc.
# Normal revolt titles for faction revolts should still have access to all normal succession laws
temporary_landless_title_trigger = {
	trigger_if = {
		limit = { temporary = yes }
		holder_scope = { is_landed = no } # major faction revolts are also landless, so this will do instead
	}
}
